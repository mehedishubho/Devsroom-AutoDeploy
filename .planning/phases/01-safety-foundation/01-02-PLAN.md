**Add verification method after the atomic swap in deploy():**

1. **Add method `verify_deployment(string $plugin_path, string $plugin_slug): array`:**
   Returns `array('success' => bool, 'message' => string, 'checks' => array())`

   Checks (in order):
   ```php
   // Check 1: Main plugin file exists
   $main_file = $plugin_path . '/' . $plugin_slug . '.php';
   if (! file_exists($main_file)) {
       return array('success' => false, 'message' => 'Main plugin file not found', 'checks' => array('file_exists' => false));
   }

   // Check 2: PHP syntax check via token_get_all()
   try {
       $code = file_get_contents($main_file);
       if ($code === false) {
           return array('success' => false, 'message' => 'Could not read main plugin file', 'checks' => array('syntax' => false));
       }
       token_get_all($code, TOKEN_PARSE);
   } catch (\Throwable $e) {
       return array('success' => false, 'message' => 'PHP syntax error: ' . $e->getMessage(), 'checks' => array('syntax' => false));
   }

   // Check 3: Plugin header exists
   $header_content = file_get_contents($main_file, false, null, 0, 8192);
   if (! preg_match('/^[\s\/*#@]*Plugin Name:/mi', $header_content)) {
       return array('success' => false, 'message' => 'Plugin header not found in main file', 'checks' => array('header' => false));
   }

   // Check 4: Critical files readable
   $critical_paths = array($main_file, $plugin_path . '/core');
   foreach ($critical_paths as $path) {
       if (file_exists($path) && ! is_readable($path)) {
           return array('success' => false, 'message' => 'Critical path not readable: ' . basename($path), 'checks' => array('readable' => false));
       }
   }

   // Check 5: Clear OPcache
   if (function_exists('wp_opcache_invalidate')) {
       wp_opcache_invalidate($main_file, true);
   }

   return array('success' => true, 'message' => 'Verification passed', 'checks' => array(
       'file_exists' => true, 'syntax' => true, 'header' => true, 'readable' => true
   ));
   ```

2. **Add verification + rollback sequence in deploy() after the atomic swap section:**

   ```php
   // Post-deploy verification
   $this->logger->info($deployment_id, 'Running post-deploy verification');
   $verification = $this->verify_deployment($plugin_path, $repository['plugin_slug']);
   $this->logger->info($deployment_id, 'Verification result', $verification);

   if (! $verification['success']) {
       // Rollback: restore .old directory
       $this->logger->warning($deployment_id, 'Verification failed, rolling back', $verification);

       $failed_path = $plugin_path . '.failed';
       // Move broken deployment aside
       @rename($plugin_path, $failed_path);

       // Restore previous version
       if (is_dir($old_path)) {
           if (! @rename($old_path, $plugin_path)) {
               // Fallback copy
               $this->copy_directory($old_path, $plugin_path);
               $this->cleanup_dir($old_path);
           }
       }

       // Clean up failed deployment
       $this->cleanup_dir($failed_path);

       return array('success' => false, 'message' => 'Deployment verification failed: ' . $verification['message']);
   }

   // Verification passed — clean up .old directory
   if (is_dir($old_path)) {
       $this->cleanup_dir($old_path);
   }
   ```
